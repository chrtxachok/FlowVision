```python
# app/services/ocr_service.py
async def process_document(
    self, file_bytes: bytes, file_type: str, document_type: str
) -> OCRResponse:
    start_ms = time.time()
    logger.info(f"Processing document (type={document_type})...")
    
    # Шаг 1: Преобработка
    logger.info("Step 1: Preprocessing image...")
    preprocess_start = time.time()
    
    ocr_result = await asyncio.get_event_loop().run_in_executor(
        None, self._run_ocr, file_bytes, file_type
    )
    
    preprocess_time = (time.time() - preprocess_start) * 1000
    logger.info(f"✓ Preprocessing done ({preprocess_time:.0f}ms)")
    logger.info(f"  Lines detected: {len(ocr_result.get('lines', []))}")
    
    # Шаг 2: Получение процессора документа
    logger.info("Step 2: Getting document processor...")
    processor = self._processors.get(document_type)
    
    if processor is None:
        logger.warning(f"⚠️  No processor for type '{document_type}'")
        extracted_data = {}
        confidence = 0.0
        warnings = [f"Unknown document_type '{document_type}'"]
    else:
        # Шаг 3: Применение процессора (regex extraction)
        logger.info(f"Step 3: Applying {document_type} processor...")
        processor_start = time.time()
        
        result = processor.process(ocr_result, file_type)
        extracted_data = result.get("extracted_data", {})
        confidence = result.get("confidence", 0.0)
        warnings = result.get("warnings", [])
        
        processor_time = (time.time() - processor_start) * 1000
        logger.info(f"✓ Processing done ({processor_time:.0f}ms)")
        logger.info(f"  Fields found: {sum(1 for v in extracted_data.values() if v.get('value'))}")
        logger.info(f"  Confidence: {confidence:.1%}")
    
    elapsed_ms = int((time.time() - start_ms) * 1000)
    logger.info(f"✓ Total processing time: {elapsed_ms}ms")
    
    return OCRResponse(
        status=ProcessingStatus.SUCCESS if extracted_data else ProcessingStatus.PARTIAL,
        document_type=document_type,
        confidence=confidence,
        processing_time_ms=elapsed_ms,
        extracted_data=extracted_data,
        raw_text=ocr_result.get("full_text", ""),
        warnings=warnings,
        metadata={
            "lines_detected": len(ocr_result.get("lines", [])),
            "file_type": file_type,
            "ocr_backend": self._backend,
        },
    )
```

```python
# app/ocr/preprocess.py
def bytes_to_numpy(file_bytes: bytes, file_type: str) -> np.ndarray:
    """Преобразование файла в numpy массив"""
    if file_type == "application/pdf":
        # Рендер первой страницы PDF
        doc = fitz.open(stream=file_bytes, filetype="pdf")
        pix = doc[0].get_pixmap()
        img_bgr = cv2.cvtColor(np.array(pix.samples), cv2.COLOR_RGB2BGR)
    else:
        # Загрузка JPG/PNG
        img_bgr = cv2.imdecode(np.frombuffer(file_bytes, np.uint8), cv2.IMREAD_COLOR)
    
    return img_bgr

def enhance_image(img: np.ndarray) -> np.ndarray:
    """Улучшение качества изображения"""
    # 1. CLAHE (Contrast Limited Adaptive Histogram Equalization)
    clahe = cv2.createCLAHE(clipLimit=2.0, tileGridSize=(8, 8))
    enhanced = clahe.apply(cv2.cvtColor(img, cv2.COLOR_BGR2GRAY))
    
    # 2. Adaptive binarization
    binary = cv2.adaptiveThreshold(enhanced, 255, cv2.ADAPTIVE_THRESH_GAUSSIAN_C,
                                   cv2.THRESH_BINARY, 11, 2)
    
    # 3. Morphological operations (denoise)
    kernel = cv2.getStructuringElement(cv2.MORPH_RECT, (3, 3))
    denoised = cv2.morphologyEx(binary, cv2.MORPH_CLOSE, kernel)
    
    return denoised
```

```python
# app/ocr/pipeline.py
def _init_easyocr(lang: str, device: str, languages: List[str] | None):
    """Инициализация EasyOCR Reader"""
    langs = languages or [lang]
    gpu = device == "gpu"
    logger.info(f"Initializing EasyOCR (langs={','.join(langs)}, gpu={gpu})...")
    reader = easyocr.Reader(langs, gpu=gpu)
    return reader

def run(self, image: np.ndarray):
    """Запуск OCR на изображении"""
    logger.info("Running EasyOCR recognition...")
    
    # EasyOCR ожидает RGB, конвертируем из BGR
    img_rgb = cv2.cvtColor(image, cv2.COLOR_BGR2RGB)
    
    # Распознавание
    results = self._ocr.readtext(img_rgb)
    
    # Результат: [(bbox, text, confidence), ...]
    # bbox = [[x1, y1], [x2, y2], [x3, y3], [x4, y4]]
    
    logger.info(f"✓ Recognized {len(results)} text lines")
    for i, (bbox, text, conf) in enumerate(results[:5]):
        logger.info(f"  Line {i}: '{text}' ({conf:.2f})")
    
    return results
```

```python
async def process_document(
    file: UploadFile = File(...),
    document_type: str = "waybill",

    """
    Основной HTTP эндпоинт.

    - Парсинг multipart/form-data
    - Валидацию входящих данных
    - Обработку ошибок на уровне HTTP
    """
    try:
        # Валидация
        if api_key != settings.API_KEY:
            raise HTTPException(status_code=401, detail="Invalid API key")
        
        if file.content_type not in ["image/jpeg", "image/png", "application/pdf"]:
            raise HTTPException(status_code=400, detail="Unsupported file type")
        
        if file.size > settings.MAX_FILE_SIZE:

        # Читаем файл

            file_bytes=file_bytes,
            file_type=file.content_type,
            document_type=document_type
        )
        
        return result

        raise

        logger.error(f"OCR processing error: {e}")
```

```python
class OCRService:
    """
    Точка входа для обработки.
    Отвечает за:
    - Выбор OCR backend'а
    - Оркестрирование pipeline
    - Выбор процессора документа
    """
    
    async def process_document(self, file_bytes, file_type, document_type):
        start_ms = time.time()
        
        # Выбираем backend
        if self._backend == "donut":
            extracted_data, raw_text, confidence, warnings = \
                self._run_donut(file_bytes, file_type, document_type)
        else:
            ocr_result = self._run_ocr(file_bytes, file_type)
            processor = self._processors.get(document_type)
            
            result = processor.process(ocr_result, file_type)
            extracted_data = result.get("extracted_data", {})
            confidence = result.get("confidence", 0.0)
            warnings = result.get("warnings", [])
        
        elapsed_ms = int((time.time() - start_ms) * 1000)
        
        return OCRResponse(
            status=ProcessingStatus.SUCCESS,
            document_type=document_type,
            confidence=confidence,
            processing_time_ms=elapsed_ms,
            extracted_data=extracted_data,
            raw_text=raw_text,
            warnings=warnings,
            metadata={...}
        )
```

```python
class OCRPipeline:
    """
    OCR pipeline.
    Отвечает за:
    - Инициализацию EasyOCR/PaddleOCR
    - Запуск распознавания текста
    """
    
    def run(self, image: np.ndarray) -> List[Any]:
        """Запустить OCR на изображении"""

            # EasyOCR ожидает RGB
            img_rgb = cv2.cvtColor(image, cv2.COLOR_BGR2RGB)
            results = self._ocr.readtext(img_rgb)
        else:
            # PaddleOCR работает с BGR
            results = self._ocr.ocr(image)
        
        return results
```

```javascript
// 1. Компрессия изображения перед отправкой (опционально)
async function compressImage(file) {
    return new Promise((resolve) => {
        const reader = new FileReader();
        reader.onload = (e) => {
            const img = new Image();
            img.onload = () => {
                const canvas = document.createElement('canvas');
                const ctx = canvas.getContext('2d');
                
                // Масштабируем если > 1920px
                let width = img.width;
                let height = img.height;
                if (width > 1920) {
                    height = (height * 1920) / width;
                    width = 1920;
                }
                
                canvas.width = width;
                canvas.height = height;
                ctx.drawImage(img, 0, 0, width, height);
                
                canvas.toBlob(resolve, 'image/jpeg', 0.8);
            };
            img.src = e.target.result;
        };
        reader.readAsDataURL(file);
```
